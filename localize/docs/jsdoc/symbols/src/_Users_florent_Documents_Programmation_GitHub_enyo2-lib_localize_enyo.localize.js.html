  6 enyo.kind({
  7 	name: "enyo.localize",
  8 	kind: enyo.Object,
  9 	/**
 10 	 * Translation Module for Enyo
 11 	 * @name enyo.localize
 12 	 * @namespace
 13  	 * @version 2 (02/03/2012)
 14 	 * @author MacFJA
 15 	 * @type Object
 16 	 */
 17 	statics: {
 18 		/** @lends enyo.localize */
 19 		/**
 20 		 * The source language (the developper language)
 21 		 * @type String
 22 		 * @default "en_US"
 23 		*/
 24 		source: "en_US",
 25 		/**
 26 		 * The destination language
 27 		 * @type String
 28 		 * @default "fr_FR"
 29 		*/
 30 		destination: "fr_FR",
 31 	
 32 		/**
 33 		 * Internal attribute that contains all translations
 34 		 * @private
 35 		 * @type Object[]
 36 		*/
 37 		list: [],
 38 
 39 		/**
 40 		 * Return the list of available translation
 41 		 * @returns List of translation
 42 		 * @type Object[]
 43 		 */
 44 		getList : function() {
 45 			return this.list;
 46 		},
 47 	
 48 		/**
 49 		 * Get user language from the navigator (for destination language)
 50 		 * @returns Return <tt><strong>true</strong></tt> is a language is found, <tt><strong>false</strong></tt> otherwise
 51 		 * @type Boolean
 52 		 */
 53 		autoDetectDestination: function() {//Get language from navigator (for destination language)
 54 			if (navigator.browserLanguage)
 55 				{ var language = navigator.browserLanguage; }//IE case :X
 56 			else if(navigator.language)
 57 				{ var language = navigator.language; }//Modern navigator case
 58 			else {//An old navigator
 59 				return false;
 60 			}
 61 
 62 			if(language.length == 2) {
 63 				//only 2 letter format (i.e. "en", "fr", "es", "de", etc.)
 64 				this.destination = language.toLowerCase();
 65 				return true;
 66 			}
 67 			else if(language.length == 5 && language.indexOf('-') > -1) {
 68 				//Standard navigator format (i.e. "en-us", "en-uk", "fr-fr", "fr-ca", "fr-be", etc.)
 69 				this.destination = language.substr(0,2)+"_"+language.substr(3,3).toUpperCase();//set format from "xx-yy" to "xx_YY"
 70 				return true;
 71 			}
 72 			else {//Unrecognized format
 73 				return false;
 74 			}
 75 		},
 76 	
 77 		/**
 78 		 * Add a group of translation
 79 		 * @param {Object[]} newList The list of new translation
 80 		 */
 81 		addArray: function(newList) {
 82 			for (var tour=0; tour < newList.length; tour++) {
 83 				this.addLocalization(newList[tour]);
 84 			};
 85 		},
 86 	
 87 		/**
 88 		 * Add a translation to the list
 89 		 * @param {Object} item The new translation
 90 		 */
 91 		addLocalization: function(item) {
 92 			this.list.push(item);
 93 		},
 94 	
 95 		/**
 96 		 * Search and select the right plural form of a given text
 97 		 * @private
 98 		 * @param {String} data The give text
 99 		 * @param {Object} [context] JSON object that contains variables
100 		 * @returns The transformed text
101 		 * @type String
102 		*/
103 		pluralSearch: function(data, context) {
104 			var re = /\{\$\w+\|.[^}]+\}/g;//RegEx search for {$something|another thing}
105 			var search = re.exec(data);
106 
107 			if(search == null)//Plural don't exist
108 				{ return data; }
109 
110 			var plural = search[0];
111 
112 			//Separate base and each case
113 			var list = plural.split("|");
114 
115 			//Remove the first "{"
116 			list[0] = list[0].substring(1);
117 			//Remove the last "}"
118 			var item = list[list.length-1];
119 			list[list.length-1] = item.substring(0, item.length-1);
120 
121 			//Get the context value (set to "*" if no context found)
122 			if(context == undefined || context.length < 1)
123 				{ var cValue = "*"; }
124 			else
125 				{ var cValue = (context[list[0].substring(1)] == undefined)?"*":context[list[0].substring(1)]; }
126 
127 
128 			//Extract the number
129 			for (var tour=1; tour < list.length; tour++) {
130 				var infos = list[tour].split(":");
131 				list[tour] = {"id": infos[0], "text": infos[1]};
132 			};
133 
134 			//Replace $obj by {$obj} in all fields
135 			var re = new RegExp("\\"+list[0], "g");
136 			for (var tour=1; tour < list.length; tour++) {
137 				var item = list[tour].text;
138 				list[tour].text = item.replace(re, "{"+list[0]+"}");
139 			};
140 
141 			//Select the good choice
142 			var choice = "";
143 			for (var tour=1; tour < list.length; tour++) {
144 				if(list[tour].id == cValue) {
145 					choice = list[tour].text;
146 				}
147 			};
148 
149 			if(choice == "") {
150 				//If we are here, then search for "*" choice
151 				for (var tour=1; tour < list.length; tour++) {
152 					if(list[tour].id == "*") {
153 						choice = list[tour].text;
154 					}
155 				};
156 			}
157 			//Update the source
158 			data = data.substring(0, data.indexOf(plural))+choice+data.substring(data.indexOf(plural)+plural.length);
159 
160 			return this.pluralSearch(data, context);
161 		}
162 	}
163 })

165 /**
166  * Search, translate and apply transformation to a text
167  * @param {String} source The text to translate
168  * @param {Object} [context] JSON object that contains variables
169  * @returns The translated text
170  * @type String
171  * @requires <tt><strong>enyo.localize</strong></tt>
172  */
173 enyo.T = function(source, context) {
174 	var longSource = enyo.localize.source;//Long language format ("xx_YY")
175 	var shortSource = longSource.substr(0,2);//Short language format ("xx")
176 	
177 	var longDest = enyo.localize.destination;//Long language format ("xx_YY")
178 	var shortDest = longDest.substr(0,2);//Short language format ("xx")
179 	
180 	var resultat = source;
181 	
182 	for(var tour in enyo.localize.getList()) {
183 		var item = enyo.localize.getList()[tour];
184 		//Search source translation
185 		if(
186 			source == item[longSource] ||
187 			source == item[shortSource] ||
188 			source == item['key']
189 		) {//In Long or short format or special key
190 			//Select the good translation
191 			if(item[longDest] != undefined) {//in long language if exist
192 				resultat = item[longDest];
193 			}
194 			else if(item[shortDest] != undefined) {//in short if exist and long don't
195 				resultat = item[shortDest];
196 			}
197 			else if(item['default'] != undefined){//No long or short format language, the language don't have translation
198 				resultat = item['default'];
199 			}
200 			if(resultat != source)//If resultat different than source, the translation have been found
201 				{ continue; }
202 		}
203 	}
204 	
205 	resultat = enyo.localize.pluralSearch(resultat, context);//Search and transform plural form
206 	return enyo.macroize(resultat, context);
207 }
