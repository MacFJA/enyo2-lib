  7 /**
  8  * @name onyx.TextArea
  9  * @class
 10  * @requires onyx.Input
 11  * @extends onyx.Input
 12  */
 13 enyo.kind({
 14 	/** @lends onyx.TextArea */
 15 	name: "onyx.TextArea",
 16 	kind: "onyx.Input",
 17 
 18 	published: {
 19 		/** @lends onyx.TextArea# */
 20 		/**
 21 		 * Define the resize behavoir (use getter/setter)
 22 		 * @type Boolean
 23 		 * @default true
 24 		 */
 25 		autoResize: true,
 26 		
 27 		/**
 28 		 * The minimum number of row to display (use getter/setter)
 29 		 * @type Number
 30 		 * @default 2
 31 		 */
 32 		minRows: 2
 33 	},
 34 
 35 	events: {
 36 		/*a* @lends onyx.TextArea# */
 37 		/**
 38 		 * Inform that the object size has changed
 39 		 * @event
 40 		 * @param {Object} inSender Event's sender
 41 		 * @param {Number} oldSize The old number of rows
 42 		 * @param {Number} newSize The new number of rows
 43 		 * @see enyojs.com for more information about events
 44 		 */
 45 		onSizeChange: ""
 46 	},
 47 	/**
 48 	 * The node tag name
 49 	 * @private
 50 	 * @type String
 51 	 */
 52 	tag: "textarea",
 53 
 54 	/**
 55 	 * Attribute that contain the actual number of rows
 56 	 * @private
 57 	 * @type Number
 58 	 */
 59 	rowsCount: 2,
 60 	
 61 	/**
 62 	 * create function, init the object
 63 	 * @private
 64 	 */
 65 	create: function() {
 66 		this.inherited(arguments);
 67 		this.doResize();
 68 	},
 69 	/**
 70 	 * Do the resize calculation and update the number of row
 71 	 * @private
 72 	 */
 73 	doResize: function() {
 74 		var lastRowsCount = this.rowsCount;
 75 		this.setAttribute("rows", this.minRows);
 76 		this.rowsCount = this.minRows;
 77 		
 78 		if(!this.hasNode() || !this.autoResize)
 79 		{
 80 			if(this.rowsCount != lastRowsCount)
 81 				{ this.doSizeChange({before: lastRowsCount, now: this.rowsCount}); }
 82 			return;
 83 		}
 84 
 85 		var node = this.node;
 86 
 87 		while(node.clientHeight < node.scrollHeight) {
 88 			this.rowsCount++;
 89 			this.setAttribute("rows", this.rowsCount);
 90 		}
 91 
 92 		if(this.rowsCount != lastRowsCount)
 93 			{ this.doSizeChange(lastRowsCount, this.rowsCount); }
 94 	},
 95 	/**
 96 	 * Handler for <q>minRows</q> value change
 97 	 * @private
 98 	 */
 99 	minRowsChanged: function() {
100 		if(this.minRows < 1)
101 			{ this.minRows = 1; }
102 		this.doResize();
103 	},
104 	/**
105 	 * Handler for <q>value</q> value change
106 	 * @private
107 	 */
108 	valueChanged: function() {
109 		this.inherited(arguments);
110 		this.doResize();
111 	},
112 	/**
113 	 * Handler for <q>onyx.Input</q> <q>oninput</q> events, call <q>doResize</q>
114 	 * @private
115 	 */
116 	input: function() {
117 		this.inherited(arguments);
118 		this.doResize();
119 	}
120 });
