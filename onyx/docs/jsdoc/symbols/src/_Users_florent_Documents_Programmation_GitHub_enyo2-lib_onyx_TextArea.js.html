 15 /**
 16  * @name onyx.TextArea
 17  * @class
 18  * @requires onyx.Input
 19  * @extends onyx.Input
 20  * @version 1.1 (03/03/2012)
 21  */
 22 enyo.kind({
 23 	/** @lends onyx.TextArea */
 24 	name: "onyx.TextArea",
 25 	kind: "onyx.Input",
 26 
 27 	published: {
 28 		/** @lends onyx.TextArea# */
 29 		/**
 30 		 * Define the resize behavoir (use getter/setter)
 31 		 * @type Boolean
 32 		 * @default true
 33 		 */
 34 		autoResize: true,
 35 		
 36 		/**
 37 		 * The minimum number of row to display (use getter/setter)
 38 		 * @type Number
 39 		 * @default 2
 40 		 */
 41 		minRows: 2,
 42 		
 43 		/**
 44 		 * The maximum number of row to display (use getter/setter).<br />
 45 		 * Set to <tt><b>null</b></tt> to don't limit.
 46 		 * @type Number
 47 		 * @default null
 48 		 */
 49 		maxRows: null
 50 	},
 51 
 52 	events: {
 53 		/** @lends onyx.TextArea# */
 54 		/**
 55 		 * Inform that the object size has changed
 56 		 * @event
 57 		 * @param {Object} inSender Event's sender
 58 		 * @param {Number} oldSize The old number of rows
 59 		 * @param {Number} newSize The new number of rows
 60 		 * @see enyojs.com for more information about events
 61 		 */
 62 		onSizeChange: ""
 63 	},
 64 	/**
 65 	 * The node tag name
 66 	 * @private
 67 	 * @type String
 68 	 */
 69 	tag: "textarea",
 70 
 71 	/**
 72 	 * Attribute that contain the actual number of rows
 73 	 * @private
 74 	 * @type Number
 75 	 */
 76 	rowsCount: 2,
 77 	
 78 	/**
 79 	 * create function, init the object
 80 	 * @private
 81 	 */
 82 	create: function() {
 83 		this.inherited(arguments);
 84 		this.doResize();
 85 	},
 86 	/**
 87 	 * Do the resize calculation and update the number of row
 88 	 * @private
 89 	 */
 90 	doResize: function() {
 91 		var lastRowsCount = this.rowsCount;
 92 		this.setAttribute("rows", this.minRows);
 93 		this.rowsCount = this.minRows;
 94 		
 95 		if(!this.hasNode() || !this.autoResize)
 96 		{
 97 			if(this.rowsCount != lastRowsCount)
 98 				{ this.doSizeChange({before: lastRowsCount, now: this.rowsCount}); }
 99 			return;
100 		}
101 
102 		var node = this.node;
103 
104 		while(
105 			node.clientHeight < node.scrollHeight &&
106 			(!this.maxRows || this.rowsCount < this.maxRows)
107 		) {
108 			this.rowsCount++;
109 			this.setAttribute("rows", this.rowsCount);
110 		}
111 
112 		if(this.rowsCount != lastRowsCount)
113 			{ this.doSizeChange(lastRowsCount, this.rowsCount); }
114 	},
115 	/**
116 	 * Handler for <q>minRows</q> value change
117 	 * @private
118 	 */
119 	minRowsChanged: function() {
120 		if(this.minRows < 1)
121 			{ this.minRows = 1; }
122 		this.doResize();
123 	},
124 	/**
125 	 * Handler dor <q>maxRows</1> value change
126 	 * @private
127 	 */
128 	maxRowsChanged: function() {
129 		if(this.maxRows < this.minRows)
130 			{ this.maxRows = this.minRows; }
131 		this.doResize();
132 	},
133 	/**
134 	 * Handler for <q>value</q> value change
135 	 * @private
136 	 */
137 	valueChanged: function() {
138 		this.inherited(arguments);
139 		this.doResize();
140 	},
141 	/**
142 	 * Handler for <q>onyx.Input</q> <q>oninput</q> events, call <q>doResize</q>
143 	 * @private
144 	 */
145 	input: function() {
146 		this.inherited(arguments);
147 		this.doResize();
148 	}
149 });
