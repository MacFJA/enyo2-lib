1 /**
  2  * Enyo UI

  7 
  8 /**
  9  * Level progess bar
 10  * @name onyx.LevelProgress
 11  * @class
 12  * @author MacFJA
 13  * @version 1.0 (24/03/2012)
 14  */
 15 enyo.kind({
 16 	name: "onyx.LevelProgress",
 17 	kind: "enyo.Control",
 18 	classes: "enyo-unselectable onyx-level-bar",
 19 
 20 	published: {
 21 		/** @lends onyx.LevelProgress# */
 22 
 23 		/**
 24 		 * The warning level (yellow) (use getter/setter)
 25 		 * @type Number
 26 		 * @default 5
 27 		 */
 28 		warning: 5,
 29 
 30 		/**
 31 		 * The critical level (red) (use getter/setter)
 32 		 * @type Number
 33 		 * @default 3
 34 		 */
 35 		critical: 3,
 36 		
 37 		/**
 38 		 * The maximum value of the progression (use getter/setter)
 39 		 * @type Number
 40 		 * @default 10
 41 		 */
 42 		maximum: 10,
 43 		
 44 		/**
 45 		 * The minimum value of the progression (use getter/setter)
 46 		 * @type Number
 47 		 * @default 0
 48 		 */
 49 		minimum: 0,
 50 		
 51 		/**
 52 		 * The value of the progression (use getter/setter)
 53 		 * @type Number
 54 		 * @default 5
 55 		 */
 56 		value: 5,
 57 		
 58 		/**
 59 		 * Define is user can change value (use getter/setter)
 60 		 * @type Boolean
 61 		 * @default true
 62 		 */
 63 		editable: true,
 64 		
 65 		/**
 66 		 * Display the value of each level (use getter/setter)
 67 		 * @type Boolean
 68 		 * @default true
 69 		 */
 70 		showValue: true,
 71 	},
 72 
 73 	/** @private */
 74 	handlers: {
 75 		onresize: "rezise",
 76 		ontap: "tap",
 77 		ondragstart: "dragstart",
 78 		ondragfinish: "dragfinish",
 79 		onenter: "enter"
 80 	},
 81 
 82 	events: {
 83 		/** @lends onyx.LevelProgress# */
 84 		/**
 85 		 * Inform that the value has changed
 86 		 * @event
 87 		 * @param {Object} inSender Event's sender
 88 		 * @param {Object} inData <tt>inData.value</tt>, contain the new value
 89 		 * @see enyojs.com for more information about events
 90 		 */
 91 		onChange: "",
 92 		/**
 93 		 * Inform that the value is changing
 94 		 * @event
 95 		 * @param {Object} inSender Event's sender
 96 		 * @param {Object} inData <tt>inData.value</tt>, contain the new value
 97 		 * @see enyojs.com for more information about events
 98 		 */
 99 		onChanging: ""
100 	},
101 	
102 	margin: 6,//Left + Right
103 	inDrag: false,//is the drag occurs?
104 	
105 	/** @lends onyx.LevelProgress# */
106 	
107 	/**
108 	 * create function, init the object
109 	 * @private
110 	 */
111 	create: function() {
112 		this.inherited(arguments);
113 		enyo.asyncMethod(this, "redraw", true, true);
114 	},
115 
116 	/**
117 	 * Handler for <q>warning</q> value change
118 	 * @private
119 	 */
120 	  warningChanged: function() {
121 		this.redraw();
122 	},
123 	/**
124 	 * Handler for <q>critical</q> value change
125 	 * @private
126 	 */
127 	 criticalChanged: function() {
128 		this.redraw();
129 	},
130 	/**
131 	 * Handler for <q>maximum</q> value change
132 	 * @private
133 	 */
134 	  maximumChanged: function() {
135 		if(this.maximum <= this.minimum) {
136 			this.maximum = this.minimum+1;
137 		}
138 		this.redraw(true, true);
139 	},
140 	/**
141 	 * Handler for <q>minimum</q> value change
142 	 * @private
143 	 */
144 	  minimumChanged: function() {
145 		if(this.minimum >= this.maximum) {
146 			this.minimum = this.maximum-1;
147 		}
148 		this.redraw(true, true);
149 	},
150 	/**
151 	 * Handler for <q>value</q> value change
152 	 * @private
153 	 */
154 	    valueChanged: function() {
155 		if(this.value < this.minimum) { this.value = this.minimum; }
156 		if(this.value > this.maximum) { this.value = this.maximum; }
157 		this.redraw();
158 	},
159 	/**
160 	 * Handler for <q>showValue</q> value change
161 	 * @private
162 	 */
163 	showValueChanged: function() {
164 		this.redraw();
165 	},
166 
167 	/**
168 	 * Function that destroy all level and recreate them
169 	 * @private
170 	 */
171 	build: function() {
172 		this.destroyComponents();
173 		this.createComponent({
174 			kind: "enyo.Control",
175 			onenter: "enter",
176 			classes: "onyx-level-element-hidden",
177 			value: this.minimum,
178 			type: "visual"
179 		});
180 		for(var tour=this.minimum+1;tour<=this.maximum;tour++) {
181 			this.createComponent({
182 				kind: "enyo.Control",
183 				onenter: "enter",
184 				classes: "onyx-level-element",
185 				value: tour,
186 				type: "visual"
187 			});
188 		}
189 		this.render();
190 	},
191 
192 	/**
193 	 * Function that redaw (update visual) all level
194 	 * @function
195 	 * @param {Boolean} updateWidth inform that the with of levels need to be recalculate
196 	 * @param {Boolean} force inform that the all levels need to be recreate
197 	 */
198 	redraw: function(updateWidth, force) {
199 		if(force) { this.build(); }
200 		
201 		var elements = this.getComponents();
202 
203 		var bounds = this.getBounds()
204 		
205 		var elementWidth = Math.floor(bounds.width/(this.maximum-this.minimum)-this.margin);//-2 cause by border
206 		var className = "normal";
207 		if(this.value > this.warning)       { className = "normal"; }
208 		else if(this.value > this.critical) { className = "warning"; }
209 		else                                { className = "critical"; }
210 		
211 		for(tour=0;tour<elements.length;tour++) {
212 			if(updateWidth || force) { elements[tour].setBounds({width: elementWidth}, 'px'); }
213 			var elmValue = elements[tour].value;
214 			var isOn = elmValue <= this.value;
215 			elements[tour].setContent(this.showValue?elmValue:"");
216 			elements[tour].addRemoveClass("normal",   isOn && className == "normal");
217 			elements[tour].addRemoveClass("warning",  isOn && className == "warning");
218 			elements[tour].addRemoveClass("critical", isOn && className == "critical");
219 			elements[tour].addRemoveClass("off", !isOn);
220 		}
221 	},
222 	
223 	/**
224 	 * Handler for <q>onResize</q> event
225 	 * @private
226 	 */
227 	rezise: function() {
228 		this.redraw(true);
229 	},
230 	/**
231 	 * Handler for <q>onTap</q> event
232 	 * @private
233 	 */
234 	tap: function(inSender, inEvent) {
235 		if(!this.editable) { return; }
236 		if(inSender.type && inSender.type == "visual") {
237 			this.setValue(inSender.value);
238 			this.doChange({value: this.value});
239 		}
240 	},
241 	/**
242 	 * Handler for <q>onDragStart</q> event
243 	 * @private
244 	 */
245 	dragstart: function(inSender, inEvent) {
246 		if (this.editable && inEvent.horizontal) {
247 			this.inDrag = true;
248 			if(inSender.type && inSender.type == "visual") {
249 				this.setValue(inSender.value);
250 				this.doChanging({value: this.value});
251 			}
252 			inEvent.preventNativeDefault();
253 			return true;
254 		}
255 	},
256 	/**
257 	 * Handler for <q>onDragFinish</q> event
258 	 * @private
259 	 */
260 	dragfinish: function(inSender, inEvent) {
261 		this.inDrag = false;
262 		inEvent.preventTap();
263 		this.doChange({value: this.value});
264 		return true;
265 	},
266 	/**
267 	 * Handler for <q>onEnter</q> event
268 	 * @private
269 	 */
270 	enter: function(inSender, inEvent) {
271 		if(!this.editable || !this.inDrag) { return; }
272 		if(inSender.type && inSender.type == "visual") {
273 			this.setValue(inSender.value);
274 			this.doChanging({value: this.value});
275 			return true;
276 		}
277 	}
278 });
