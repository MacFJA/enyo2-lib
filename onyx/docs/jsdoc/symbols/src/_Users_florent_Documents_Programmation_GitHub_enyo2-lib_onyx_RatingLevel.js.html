  8 /**
  9  * Rating level selector
 10  * @name onyx.RatingLevel
 11  * @class
 12  * @author MacFJA
 13  * @version 1.0 (24/03/2012)
 14  * @extends onyx.LevelProgress
 15  */
 16 enyo.kind({
 17 	name: "onyx.RatingLevel",
 18 	kind: "onyx.LevelProgress",
 19 	classes: "enyo-unselectable onyx-rating-bar",
 20 
 21 	
 22 	/**
 23 	 * create function, init the object
 24 	 * @private
 25 	 */
 26 	create: function() {
 27 		this.margin = 7;
 28 		this.inherited(arguments);
 29 	},
 30 
 31 	/**
 32 	 * Function that destroy all level and recreate them
 33 	 * @private
 34 	 */
 35 	build: function() {
 36 		this.destroyComponents();
 37 		this.createComponent({
 38 			kind: "enyo.Control",
 39 			onenter: "enter",
 40 			classes: "onyx-level-element-hidden",
 41 			value: this.minimum,
 42 			type: "visual"
 43 		});
 44 		for(var tour=this.minimum+1;tour<=this.maximum;tour++) {
 45 			this.createComponent({
 46 				kind: "enyo.Control",
 47 				onenter: "enter",
 48 				classes: "onyx-rating-element",
 49 				value: tour,
 50 				type: "visual"
 51 			});
 52 		}
 53 		this.render();
 54 	}
 55 });
